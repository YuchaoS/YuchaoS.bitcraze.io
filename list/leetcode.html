class Solution(object):
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        empty = {}
        for i in range(len(nums)):
            if not empty.has_key(nums[i]):
            #if nums[i] not in empty.keys():
            #use this will slow down 
                empty[target-nums[i]] = i
            else:
                return[empty[nums[i]],i]

class Solution {
    public:
    vector<int> twoSum(vector<int>& nums,int target){
        map<int,int> empty;
        vector<int> result;
        for (int i=0;i<nums.size();i++){
            if(!empty.count(nums[i])){
                empty[target-nums[i]] = i;
            }
            else{
                result.push_back(empty[nums[i]]);
                result.push_back(i);
                return result;
            }
        }
        return result;
    }

class Solution(object):
    def addTwoNumbers(self, l1, l2):
        """
        :type l1: ListNode
        :type l2: ListNode
        :rtype: ListNode
        """
        resultList = ListNode()
        re = resultList
        up = 0
        while(l1 and l2):
            result = l1.val+l2.val+up
            if(result>=10):
                re.val = result%10
                up = 1
            else:
                re.val =result
                up = 0
            l1 = l1.next
            l2 = l2.next
            if(l1 and l2):
                re.next = ListNode()
                re = re.next
        while(l1):
            re.next = ListNode()
            re = re.next
            result = up+l1.val
            if(result>=10):
                re.val = result%10
                up = 1
            else:
                re.val =result
                up = 0
            l1 = l1.next
        while(l2):
            re.next = ListNode()
            re = re.next
            result = up+l2.val
            if(result>=10):
                re.val = result%10
                up = 1
            else:
                re.val =result
                up = 0
            l2 = l2.next
        while(up):
            re.next = ListNode()
            re = re.next
            re.val = 1
            up = 0
        return resultList

class Solution(object):
    def lengthOfLongestSubstring(self, s):
        """
        :type s: str
        :rtype: int
        """
        if(len(s)==1):
            return 1;
        pre = -1
        maxl = 0
        tail = 1
        while(len(s)-pre>=maxl and tail<len(s)):
            i = tail-1
            while(i>pre):
                if(s[i]==s[tail]):
                    break
                else:
                    i = i-1
            if(tail-i>=maxl):
                maxl = tail-i
            pre = i
            tail = tail+1
        return maxl
